119 /** @brief This code controls the solenoids to open and close

135 void solenoid (void* p_params)

139  const TickType_t solenoid_period = 50; // RTOS ticks (ms) between runs
140  /** TB6612FNG motor driver input pins
141  * PWMA=>D12()
142  * AIN_2=>D11()
143  * AIN_1=>D10()
144  * BIN_1=>D09()
145  * BIN_2=>D5()
146  * PWMB=>D4()
147  */
148  const int8_t Ain_sol = PB6;
149 
150  // Initialise the xLastWakeTime variable with the current time.
151  // It will be used to run the task at precise intervals
152  TickType_t xLastWakeTime = xTaskGetTickCount();
153 
154  // init variable to pull from share
155  bool sol_on;
156  for(;;)
157  {
158  // check solenoid share
159  solenoid_on.get(sol_on);
160 
161  // if share is on, turn on solenoid, leave on for 1 sec, then turn off
162  if (sol_on == 1)
163  {
164  digitalWrite(Ain_sol, HIGH);
165  delay(1000);
166  solenoid_on.put(false);
167  }
168  // if share is off, turn off solenoid
169  else
170  {
171  digitalWrite(Ain_sol, LOW);
172  }
173  // Delay task by 50 ms since task began
174  vTaskDelayUntil (&xLastWakeTime, solenoid_period);
175  }
